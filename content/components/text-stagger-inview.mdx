# Text Stagger Inview

A component that creates a staggered text animation effect when elements enter the viewport.

## Usage

```tsx
import { TextStaggerInview } from "@/registry/new-york/ui/text-stagger-inview"

export default function Page() {
  return (
    <TextStaggerInview
      text="Your text here"
      animation="fade"
      className="text-2xl font-bold"
    />
  )
}
```
